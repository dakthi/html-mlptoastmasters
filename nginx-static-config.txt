# Static HTML Nginx Config for mlptoastmasters.org.uk
# This will replace the current proxy config when ready to switch

server {
    server_name mlptoastmasters.org.uk;

    root /var/www/mlp-toastmasters;
    index index.html;

    # Enable gzip compression
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css text/xml text/javascript application/json application/javascript application/xml+rss application/rss+xml font/truetype font/opentype application/vnd.ms-fontobject image/svg+xml;

    # Cache static assets
    location ~* \.(jpg|jpeg|png|gif|ico|css|js|svg|webp|woff|woff2|ttf|eot)$ {
        expires 1y;
        add_header Cache-Control "public, immutable";
        access_log off;
    }

    # Main location block
    location / {
        try_files $uri $uri/ =404;
    }

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;

    listen 443 ssl;
    ssl_certificate /etc/letsencrypt/live/mlptoastmasters.org.uk/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/mlptoastmasters.org.uk/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
}

server {
    if ($host = mlptoastmasters.org.uk) {
        return 301 https://$host$request_uri;
    }

    server_name mlptoastmasters.org.uk;
    listen 80;
    return 404;
}

# Instructions to switch from Docker to static HTML:
# 1. Wait for all agents to complete conversion
# 2. Test the static site locally at http://localhost:8080
# 3. Push final changes to GitHub (triggers auto-deploy to VPS)
# 4. Backup current nginx config: cp /etc/nginx/sites-enabled/mlptoastmasters.org.uk /etc/nginx/sites-available/mlptoastmasters.org.uk.docker-backup
# 5. Replace nginx config: nano /etc/nginx/sites-enabled/mlptoastmasters.org.uk (paste content above)
# 6. Test nginx: nginx -t
# 7. Reload nginx: systemctl reload nginx
# 8. Optional: Stop Docker container if no longer needed: docker stop nextjs_mlptoastmasters
